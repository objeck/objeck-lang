use API.OpenAI, API.OpenAI.Chat, Collection;

class CodeReviewer {
  function : Main(args : String[]) ~ Nil {
    token  := System.IO.Filesystem.FileReader->ReadFile("api_key.txt")->Trim();
    prompt := Pair->New("user", args[0])<String, String>;

    completion := Completion->Complete("gpt-4o", prompt, token);
    if(completion <> Nil) {
      completion->GetFirstChoice()
        ->GetMessage()<String, String>
        ->GetSecond()->PrintLine();
    };
  }
}
use Web.HTTP, Collection;

class RepoStats {
  function : Main(args : String[]) ~ Nil {
    client := Http2Client->New("api.github.com");
    client->AddHeader("user-agent", "objeck-app");

    resp := client->Get("/repos/objeck/objeck-lang");
    if(resp <> Nil) {
      "Status : {$resp->GetCode()}"->PrintLine();
      resp->GetContent()->ToString()->PrintLine();
    };
    client->Close();
  }
}
use Collection;

class WordFilter {
  function : Main(args : String[]) ~ Nil {
    words := Vector->New();
    words->AddBack("compiler"); words->AddBack("jit");
    words->AddBack("runtime"); words->AddBack("gc");
    words->AddBack("lambda");  words->AddBack("arm64");

    words->Each(\^(w : String) =>
      if(w->Size() > 3) {
        "  {$w->ToUpper()}"->PrintLine();
      });
  }
}
