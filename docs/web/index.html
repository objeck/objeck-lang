use System.IO.Net;

class Server {
  function : Main(args : String[]) ~ Nil {
    server := TCPSocketServer->New(8080);
    if(server->Listen(5)) {
      while(true) {
        client := server->Accept();
        client->WriteString("HTTP/1.1 200 OK\r\nContent-Type: text/html\r\n\r\n");
        client->WriteString("<h1>Hello from Objeck!</h1>\n");
        client->Close();
      };
    };
  }
}
class Demo {
  function : Main(args : String[]) ~ Nil {
    values := [1, 2, 3, 4, 5];
    squares := values->Map(Int)->Do(\(v) => v * v);
    squares->Each(\(v) => v->PrintLine());

    names := ["Alice", "Bob", "Carol"];
    names->Each(\(name) =>
      "Hello, {$name}!"->PrintLine());
  }
}
